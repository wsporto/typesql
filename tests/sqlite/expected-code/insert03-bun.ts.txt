import type { Database } from 'bun:sqlite';

export type Insert03Params = {
	value: number | null;
}

export type Insert03Result = {
	id: number;
	value: number | null;
}

export function insert03(db: Database, params: Insert03Params): Insert03Result | null {
	const sql = `
	INSERT INTO mytable1(value) VALUES(?) RETURNING *
	`
	const rows = db.prepare(sql)
		.values(params.value);

	return rows.length > 0 ? mapArrayToInsert03Result(rows[0]) : null;
}

function mapArrayToInsert03Result(data: any) {
	const result: Insert03Result = {
		id: data[0],
		value: data[1]
	}
	return result;
}
