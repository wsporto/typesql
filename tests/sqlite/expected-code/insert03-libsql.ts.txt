import type { Client, Transaction } from '@libsql/client';

export type Insert03Params = {
	value: number | null;
}

export type Insert03Result = {
	id: number;
	value: number | null;
}

export async function insert03(client: Client | Transaction, params: Insert03Params): Promise<Insert03Result | null> {
	const sql = `
	INSERT INTO mytable1(value) VALUES(?) RETURNING *
	`
	const { rows } = await client.execute({ sql, args: [params.value] });

	return rows.length > 0 ? mapArrayToInsert03Result(rows[0]) : null;
}

function mapArrayToInsert03Result(data: any) {
	const result: Insert03Result = {
		id: data[0],
		value: data[1]
	}
	return result;
}
