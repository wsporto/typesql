import type { Database } from 'better-sqlite3';

export type Insert04Params = {
	value1: number | null;
	value2: number | null;
	value3: number | null;
}

export type Insert04Result = {
	id: number;
	value: number | null;
}

export function insert04(db: Database, params: Insert04Params): Insert04Result[] {
	const sql = `
	INSERT INTO mytable1(
		value
	) VALUES (?), (?), (?)
	RETURNING *
	`
	return db.prepare(sql)
		.raw(true)
		.all([params.value1, params.value2, params.value3])
		.map(data => mapArrayToInsert04Result(data));
}

function mapArrayToInsert04Result(data: any) {
	const result: Insert04Result = {
		id: data[0],
		value: data[1]
	}
	return result;
}
