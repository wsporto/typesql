import type { Database } from 'better-sqlite3';

export type Delete03Result = {
	id: number;
	value: number | null;
}

export function delete03(db: Database): Delete03Result[] {
	const sql = `
	DELETE FROM mytable1 WHERE id > 1 RETURNING *
	`
	return db.prepare(sql)
		.raw(true)
		.all()
		.map(data => mapArrayToDelete03Result(data));
}

function mapArrayToDelete03Result(data: any) {
	const result: Delete03Result = {
		id: data[0],
		value: data[1]
	}
	return result;
}
