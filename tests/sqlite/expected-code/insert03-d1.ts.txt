import type { D1Database } from '@cloudflare/workers-types';

export type Insert03Params = {
	value: number | null;
}

export type Insert03Result = {
	id: number;
	value: number | null;
}

export async function insert03(db: D1Database, params: Insert03Params): Promise<Insert03Result | null> {
	const sql = `
	INSERT INTO mytable1(value) VALUES(?) RETURNING *
	`
	const rows = await db.prepare(sql)
		.bind(params.value)
		.raw({ columnNames: false });

	return rows.length > 0 ? mapArrayToInsert03Result(rows[0]) : null;
}

function mapArrayToInsert03Result(data: any) {
	const result: Insert03Result = {
		id: data[0],
		value: data[1]
	}
	return result;
}
