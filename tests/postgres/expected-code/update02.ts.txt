import pg from 'pg';

export type Update02Params = {
	id: number;
}

export type Update02Result = {
	rowCount: number;
}

export async function update02(client: pg.Client | pg.Pool | pg.PoolClient, params: Update02Params): Promise<Update02Result> {
	const sql = `
	UPDATE mytable1 SET value = 42 WHERE id = $1
	`
	return client.query({ text: sql, values: [params.id] })
		.then(res => mapArrayToUpdate02Result(res));
}

function mapArrayToUpdate02Result(data: any) {
	const result: Update02Result = {
		rowCount: data.rowCount
	}
	return result;
}
